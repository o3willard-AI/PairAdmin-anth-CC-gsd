    ```typescript
    import { CopyToClipboard } from "../../wailsjs/go/services/CommandService";

    export function useWailsClipboard() {
      const copyToClipboard = async (text: string) => {
        try {
          await CopyToClipboard(text);
        } catch (err) {
          // Fallback to browser clipboard API if Wails binding fails
          // This can happen in dev mode before bindings are generated
          console.warn("Wails clipboard failed, trying browser API:", err);
          await navigator.clipboard.writeText(text);
        }
      };

      return { copyToClipboard };
    }
    ```
    Note: The exact import path for the Wails binding depends on what `wails dev` generated. Read `frontend/wailsjs/go/` directory to find the correct import path. It may be `../../wailsjs/go/main/CommandService` depending on the Go package structure. Adjust the import path accordingly.
