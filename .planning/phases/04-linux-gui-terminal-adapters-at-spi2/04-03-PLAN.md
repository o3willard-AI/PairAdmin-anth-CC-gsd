      set((state) => {
        if (state.tabs.some((t) => t.id === id)) return;
        state.tabs.push({ id, name, degraded, degradedMsg });
        if (state.tabs.length === 1 && !degraded) {
          state.activeTabId = id; // don't auto-activate degraded tabs
        }
      });
    },
